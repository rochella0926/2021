#include<stdio.h>


char upper(char ch)
{
	return ch - ('a' - 'A');
}
char lower(char ch)
{
	return ch + ('a' - 'A');
}
int main()
{
	char in[100], out[100];
	char ch;
	int i = 0;

	printf("문자열 입력 : ");
	gets_s(in);

	do {
		ch = in[i];
	
		if ('A' <= ch && 'Z' >= ch)
			out[i] =lower(ch);
		else if ('a' <= ch && 'z' >= ch)
			out[i] = upper(ch);
		else
			out[i] = ch;

		i++;
	}while (ch != '\0');
	 out[i] = '\0';

	printf("변환된 문자 ==> %s\n", out);
}
