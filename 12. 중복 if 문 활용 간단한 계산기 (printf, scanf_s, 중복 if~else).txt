#include<stdio.h>

int main()
{
	int a,b;
	char ch;

	printf("첫번째 수를 입력하세요 : ");
	scanf_s("%d", &a);
	printf("연산자를 입력하세요 : ");
	scanf_s(" %c", &ch);
	printf("두번째 수를 입력하세요 : ");
	scanf_s("%d", &b);

	if (ch == '+')
		printf("%d+%d=%d입니다. \n", a, b, a + b);
	else if (ch == '-')
		printf("%d-%d=%d입니다. \n", a, b, a - b);
	else if (ch == '*')
		printf("%d*%d=%d입니다. \n", a, b, a * b);
	else if (ch == '/')
		printf("%d/%d=%d입니다. \n", a, b, a / b);
	else if (ch == '%')
		printf("%d%%%d=%d입니다. \n", a, b, a % b);
	else
		printf("연산자를 잘못 입력했습니다. \n");
}
