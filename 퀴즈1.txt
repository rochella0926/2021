#include<stdio.h>

main()

{
	int number;

	printf("출력하고 싶은 구구단은 몇 단인가요?");
	scanf_s("%d", &number);

	printf("%d단을 출력합니다.\n", number);
	printf("***************\n");

	for (int i = 1;i <= 9; i++)
	{
		printf("%d*%d=%d\n", number, i, number * i);
	}

	printf("***************\n");
}
